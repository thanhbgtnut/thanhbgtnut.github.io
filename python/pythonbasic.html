Truy nhp n tng phn t ca List

proc = subprocess.Popen('ping 127.0.0.1', stdout=subprocess.PIPE)


while True:
   time.sleep(1)
   poll = proc.poll()    
   
   if poll == None:            # program is running
       while True:
           line = proc.stdout.readline().decode()
           
           if line:
               print(line)
           else:
               break
   else:                       # program completed
       print('Subprocess completed.')
       break

Thao tc vi Json

Th vin json cho php chuyn i gia json string v object ca python. Hai hm thng s dng l:

  • json.loads : chuyn json string thnh object
  • json.dumps : chuyn object thnh json string

V d:

import json

students = [
   { 'name' : 'Nguyn Vn An', 'address' : 'H Ni'},
   { 'name' : 'Nguyn Th Bnh', 'address' : 'TP.HCM'},
]
print(json.dumps(students))

jsonstring = '{"name" : "Nguyn Vn An", "address" : "H Ni"}'
student = json.loads(jsonstring)
print(student['name'], student['address'])

import xml.etree.ElementTree as ET

xml = """
   <students>
       <student>
           <name>Nguyn Vn An</name>
           <address>H Ni</address>
       </student>
       <student>
           <name>Nguyn Th Bnh</name>
           <address>TP.HCM</address>
       </student>
   </students>
"""

root = ET.fromstring(xml)
for student in root:    
   name = student.find('name').text
   address = student.find('address').text
   print(name, address)

Chuyn object ca Python sang xml

import xml.etree.cElementTree as ET

students = [
   { 'name' : 'Nguyn Vn An', 'address' : 'H Ni'},
   { 'name' : 'Nguyn Th Bnh', 'address' : 'TP.HCM'},
]


root = ET.Element("students")

for student in students:
   tag = ET.SubElement(root, "student")
   ET.SubElement(tag, "name").text = student['name']
   ET.SubElement(tag, "address").text = student['address']

tree = ET.ElementTree(root)
tree.write("output.xml", encoding="UTF-8")

TCP/IP Socket

Th vin socket cho php xy dng cc chng trnh kt ni theo m hnh Client/Server qua Socket

Server:

# server.py

import socket

HOST = '127.0.0.1'
PORT = 8080

s = socket.socket()
s.bind((HOST, PORT))
s.listen()

print(f'Listen at {HOST}:{PORT}')

while True:
   conn, addr = s.accept()
   print('Connected from ', addr)
   data = conn.recv(4096)
   clientID = data.decode()
   conn.send(f'Hello {clientID}'.encode())
   conn.close()

Client :

# client.py

import socket

HOST = '127.0.0.1'
PORT = 8080
CLIENT_ID = 'Client1'

s = socket.socket()

s.connect((HOST, PORT))
s.send(CLIENT_ID.encode())
data = s.recv(4096)
print(data.decode())
s.close()
